include_directories(.)
include_directories(..)

file(GLOB src *.cpp)
file(GLOB headers *.hpp)
file(GLOB forms *.ui)
file(GLOB resources *.qrc ../gui/resources/*.qrc)

#qt specific
include_directories(${CMAKE_CURRENT_BINARY_DIR})

ADD_DEFINITIONS(-std=c++17)

if(APPLE)
    set(PROGNAME firestr)
    set(MACOSX_BUNDLE_NAME "Firestr")
    set(MACOSX_BUNDLE_GUI_IDENTIFIER "com.mempko.firestr")
    set(MACOSX_BUNDLE_ICON_FILE firestr.icns)
    set(MACOSX_BUNDLE_SHORT_VERSION_STRING 0.11.1)
    set(MACOSX_BUNDLE_VERSION 0.11.1)
    set(MACOSX_BUNDLE_LONG_VERSION Version 0.11.1)
    set(MACOSX_BUNDLE_NAME firestr)
    set(MACOSX_BUNDLE_INFO_STRING firestr)
    set(MACOSX_BUNDLE_COPYRIGHT "(C) 2016 Maxim Noah Khailo")

    add_custom_command(TARGET firestr POST_BUILD
        command mkdir -p ${CMAKE_CURRENT_BINARY_DIR}/firestr.app/Contents/Resources
        command cp -p ${CMAKE_CURRENT_BINARY_DIR}/firestr.app/Contents/MacOS
        command cp *.qm ${CMAKE_CURRENT_BINARY_DIR}/firestr.app/Contents/Resources/)

    add_executable(
        firestr
        MACOSX_BUNDLE
        ${src}
        ${resources})

    #set_target_properties(firestr PROPERTIES MACOSX_BUNDLE_INFO_PLIST ${CMAKE_SOURCE_DIR}/cmake/mac_bundle.plist.in)

    set(APPS ${CMAKE_CURRENT_BINARY_DIR}/firestr.app)



elseif(CMAKE_HOST_WIN32)
    add_executable(
        firestr
        ${src}
        ${resources})
else()
    add_executable(
        firestr
        ${src}
        ${resources})
endif()

    fire_network

    fire_util
    fire_slb
    fire_lua
    Qt6::Widgets
    Qt6::Network
    Qt6::Multimedia
    ${Boost_LIBRARIES}
    ${MISC_LIBRARIES})

add_dependencies(
    firestr 
    fire_gui
    fire_conversation
    fire_messages
    fire_user
    fire_message
    fire_network
    fire_security
    fire_util
    fire_slb
    fire_lua)

install(TARGETS firestr DESTINATION bin)

if(APPLE)
elseif(CMAKE_HOST_WIN32)
